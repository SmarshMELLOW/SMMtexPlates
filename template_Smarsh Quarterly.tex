% FIGURE DEFINITIONS --------------------------------
% page 1 figure
\newcommand{\ponefig}[4]{
	\begin{figure}[h]
		\captionsetup{width=#1\textwidth}
		\centering
		\includegraphics[width=#1\textwidth]{#2}
		\caption*{\textcolor{Dark-Gray}{#3 \textit{#4}}}
	\end{figure}
}

% page 2 figures
\newcommand{\ptwofig}[4]{
	\begin{wrapfigure}{#1}{0.45\textwidth}
		\captionsetup{width=0.45\textwidth}
		\begin{center}
			\includegraphics[width=0.45\textwidth]{#2}
		\end{center}
		\caption*{\textcolor{Dark-Gray}{#3 \textit{#4}.}}
	\end{wrapfigure}
}
